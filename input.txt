# CPU Pipeline Simulator - Test Program
# Demonstrates: forwarding, load-use stall, cache hits/misses
# Pre-loaded: Mem[10]=5, Mem[20]=10, Mem[30]=15

LOAD R2, 20(R0)
ADD R3, R1, R2
SUB R4, R3, R1
ADDI R5, R0, 42
AND R6, R3, R5
